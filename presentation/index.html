Code Highlight: Canvas API

Fetching assignments with OAuth2:

async function fetchCanvasAssignments(token, courseId) {
    const response = await fetch(`https://canvas.northeastern.edu/api/v1/courses/${courseId}/assignments`, {
        headers: {
            'Authorization': `Bearer ${token}`,
            'Accept': 'application/json'
        }
    });
    if (!response.ok) throw new Error('Failed to fetch assignments');
    const data = await response.json();
    return data.map(assignment => ({
        id: assignment.id,
        name: assignment.name,
        due_at: assignment.due_at
    }));
}
